                        //å…ˆåˆ¤æ–­æ˜¯switchcaseè¯­å¥ï¼Œç„¶åæˆ–å–å€¼ä¸å¸¸é‡å¯¹æ¯”ï¼Œç„¶ååˆ¤æ–­æ˜¯å¦å·²åˆ°è¾¾
                        if (edge.getKind() == Edge.Kind.SWITCH_CASE && edge.getCaseValue() == val && !reached.contains(edge.getTarget())) {
                            // å°†æœªéå†è¿‡çš„ switch stmt æ·»åŠ åˆ°é˜Ÿåˆ—
                            stmts.add(edge.getTarget());
                        }
                    }
                } else {
                    //è‹¥æ²¡æœ‰åˆ™default
                    Stmt target = ((SwitchStmt) s).getDefaultTarget();
                    if (!reached.contains(target)) {
                        stmts.add(target);
                    }
                }
            }else{
                for (Stmt succ : cfg.getSuccsOf(s)) {
                    if (!reached.contains(succ))
                        stmts.add(succ);
                }
            }
        } else if (s instanceof AssignStmt) {
            LValue l = ((AssignStmt)s).getLValue();
            RValue r = ((AssignStmt)s).getRValue();
            boolean is_dead = false;
            if (l instanceof Var) {  //è¦åˆ¤æ–­å·¦å€¼æ˜¯å¦èƒ½è½¬åŒ–æˆå˜é‡ç±»å‹,å‰é¢æœ‰è¿‡ç±»ä¼¼é—®é¢˜
                if (!liveVars.getResult(s).contains((Var) l)) {  // ä¸æ´»è·ƒ
                    if (hasNoSideEffect(r)) {  // æ²¡æœ‰å‰¯ä½œç”¨
                        is_dead = true;
                    }
                }
            }
            if(!is_dead){
                reach.add(s);
            }
            for (Stmt succ : cfg.getSuccsOf(s)) {
                if (!reached.contains(succ)) {
                    // å°†æœªéå†è¿‡çš„åç»­ stmt æ·»åŠ åˆ°é˜Ÿåˆ—
                    stmts.add(succ);
                }
            }
        } else{
            //è®°å½•å¯åˆ°è¾¾è¯¥ stmt
            reach.add(s);
            for (Stmt succ : cfg.getSuccsOf(s)) {
                if (!reached.contains(succ))
                    stmts.add(succ);
            }
        }
    }
    for (Stmt s : ir) {
        if(!reach.contains(s)){
            deadCode.add(s);
        }
    }
    return deadCode;
}
